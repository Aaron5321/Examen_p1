CREATE TABLE bibliotec.Cliente (idCliente INT(10) NOT NULL, nombre VARCHAR(255) NOT NULL, telefono VARCHAR(15) NOT NULL, PRIMARY KEY (idCliente)) ENGINE = InnoDB; , Comision VARCHAR(5) NOT NULL , Telefono INT(10) NOT NULL , PRIMARY KEY (Id)) ENGINE = InnoDB;
CREATE TABLE bibliotec.Mascota (idMascota INT(10) NOT NULL, nombre VARCHAR(255) NOT NULL, especie VARCHAR(50) NOT NULL, idCliente INT(10) NOT NULL, PRIMARY KEY (idMascota), FOREIGN KEY (idCliente) REFERENCES bibliotec.Cliente(idCliente)) ENGINE = InnoDB;
CREATE TABLE bibliotec.Servicio (idServicio INT(10) NOT NULL, tipoServicio VARCHAR(255) NOT NULL, costo DECIMAL(10, 2) NOT NULL, PRIMARY KEY (idServicio)) ENGINE = InnoDB;
CREATE TABLE bibliotec.Cita (idCita INT(10) NOT NULL, fecha DATE NOT NULL, idCliente INT(10) NOT NULL, idMascota INT(10) NOT NULL, idServicio INT(10) NOT NULL, idMedico INT(10) NOT NULL, PRIMARY KEY (idCita), FOREIGN KEY (idCliente) REFERENCES bibliotec.Cliente(idCliente), FOREIGN KEY (idMascota) REFERENCES bibliotec.Mascota(idMascota), FOREIGN KEY (idServicio) REFERENCES bibliotec.Servicio(idServicio), FOREIGN KEY (idMedico) REFERENCES bibliotec.MedicoVeterinario(idMedico)) ENGINE = InnoDB;
CREATE TABLE bibliotec.MedicoVeterinario (idMedico INT(10) NOT NULL, nombre VARCHAR(255) NOT NULL, especialidad VARCHAR(255) NOT NULL, PRIMARY KEY (idMedico)) ENGINE = InnoDB;
